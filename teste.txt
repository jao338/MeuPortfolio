--------------------Para criar um pdf e exibir uma prévia--------------

<!DOCTYPE html>
<html>
<head>
    <title>Visualizar PDF</title>
    <script src="https://cdnjs.cloudflare.com/ajax/libs/pdf.js/2.12.313/pdf.js"></script>
</head>
<body>
    <canvas id="pdfCanvas"></canvas>
    <script>
        // Variáveis para configurar o PDF.js
        var pdfURL = '/caminho-para-o-seu-arquivo.pdf';
        var canvas = document.getElementById('pdfCanvas');
        
        // Carregar e renderizar o PDF
        pdfjsLib.getDocument(pdfURL).promise.then(function(pdfDoc) {
            pdfDoc.getPage(1).then(function(page) {
                var scale = 1.5;
                var viewport = page.getViewport({ scale: scale });
                var context = canvas.getContext('2d');
                canvas.height = viewport.height;
                canvas.width = viewport.width;

                var renderContext = {
                    canvasContext: context,
                    viewport: viewport
                };
                
                page.render(renderContext);
            });
        });
    </script>
</body>
</html>

--------------Para criar um efeito de transição suave entre as sections-------------------

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Transição Suave entre Seções</title>
    <style>
        body {
            margin: 0;
            padding: 0;
        }
        section {
            height: 100vh;
            display: flex;
            align-items: center;
            justify-content: center;
            font-size: 2rem;
        }
    </style>
</head>
<body>
    <nav>
        <ul>
            <li><a href="#section1">Seção 1</a></li>
            <li><a href="#section2">Seção 2</a></li>
            <li><a href="#section3">Seção 3</a></li>
        </ul>
    </nav>

    <section id="section1" style="background-color: lightblue;">Seção 1</section>
    <section id="section2" style="background-color: lightgreen;">Seção 2</section>
    <section id="section3" style="background-color: lightcoral;">Seção 3</section>

    <script>
        // Intercepta os cliques nos links de navegação
        document.querySelectorAll('nav a').forEach(anchor => {
            anchor.addEventListener('click', function(e) {
                e.preventDefault();

                // Obtém o alvo da seção a partir do atributo href do link
                const targetId = this.getAttribute('href').substring(1);
                const targetSection = document.getElementById(targetId);

                // Rola suavemente para a seção alvo
                targetSection.scrollIntoView({
                    behavior: 'smooth'
                });
            });
        });
    </script>
</body>
</html>
