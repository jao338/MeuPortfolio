--------------------Para criar um pdf e exibir uma prévia--------------

<!DOCTYPE html>
<html>
<head>
    <title>Visualizar PDF</title>
    <script src="https://cdnjs.cloudflare.com/ajax/libs/pdf.js/2.12.313/pdf.js"></script>
</head>
<body>
    <canvas id="pdfCanvas"></canvas>
    <script>
        // Variáveis para configurar o PDF.js
        var pdfURL = '/caminho-para-o-seu-arquivo.pdf';
        var canvas = document.getElementById('pdfCanvas');
        
        // Carregar e renderizar o PDF
        pdfjsLib.getDocument(pdfURL).promise.then(function(pdfDoc) {
            pdfDoc.getPage(1).then(function(page) {
                var scale = 1.5;
                var viewport = page.getViewport({ scale: scale });
                var context = canvas.getContext('2d');
                canvas.height = viewport.height;
                canvas.width = viewport.width;

                var renderContext = {
                    canvasContext: context,
                    viewport: viewport
                };
                
                page.render(renderContext);
            });
        });
    </script>
</body>
</html>
